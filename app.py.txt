FRAME_ID_PATTERN = re.compile(r"f_(\d+)\.(?:jpg|jpeg|png)$", re.IGNORECASE)
TRACK_SOURCE_PATTERN = re.compile(r"(?:^|;)track_source=([^;]+)")


def _init_session_state() -> None:
    defaults: dict[str, Any] = {
        "frames_json_path": "",
        "segments_csv_path": "",
        "perception_json_path": "",
        "report_json_paths": [],
        "resolved_video_path": "",
        "uploaded_file_sig": "",
        "detecttrack_result": {},
    }
    for key, value in defaults.items():
        if key not in st.session_state:
            st.session_state[key] = value


def _frame_caption(path: str) -> str:
    name = Path(path).name
    match = FRAME_ID_PATTERN.search(name)
    if not match:
        return name
    return f"frame {int(match.group(1))} | {name}"


def _frame_idx(path: str) -> int | None:
    match = FRAME_ID_PATTERN.search(Path(path).name)
    if not match:
        return None
    return int(match.group(1))


def _extract_track_source(report_json: dict[str, Any]) -> str | None:
    audit = report_json.get("audit", {})
    if not isinstance(audit, dict):
        return None
    notes = str(audit.get("notes", ""))
    match = TRACK_SOURCE_PATTERN.search(notes)
    if not match:
        return None
    return match.group(1).strip() or None


def _parse_audit_notes_pairs(notes: str) -> list[dict[str, str]]:
    rows: list[dict[str, str]] = []
    for part in str(notes).split(";"):
        piece = part.strip()
        if not piece:
            continue
        if "=" in piece:
            key, value = piece.split("=", 1)
            rows.append({"key": key.strip(), "value": value.strip()})
        else:
            rows.append({"key": "note", "value": piece})
    return rows


def _render_video(path: str) -> None:

    video_path = Path(path)
    if not video_path.exists() or not video_path.is_file():
        st.info("Video file not found.")
        return
    if video_path.stat().st_size <= 0:
        st.warning("Video file is empty.")
        return

    try:
        video_bytes = video_path.read_bytes()
        suffix = video_path.suffix.lower()
        fmt = "video/mp4" if suffix == ".mp4" else ("video/webm" if suffix == ".webm" else None)
        if fmt:
            st.video(video_bytes, format=fmt)
        else:
            st.video(video_bytes)
    except Exception:
        st.video(str(video_path))


def _resolve_video_path(input_mode: str, path_input: str, uploaded_file: Any) -> str:
    if input_mode == "Use existing local file path":
        return services.resolve_video_input(path_input)

    if uploaded_file is None:
        raise ValueError("Please upload a video file first.")

    size = getattr(uploaded_file, "size", "unknown")
    file_sig = f"{uploaded_file.name}:{size}"
    if st.session_state.get("uploaded_file_sig") != file_sig:
        resolved = services.resolve_video_input(uploaded_file)
        st.session_state["resolved_video_path"] = resolved
        st.session_state["uploaded_file_sig"] = file_sig

    resolved_path = st.session_state.get("resolved_video_path", "")
    if not resolved_path:
        raise ValueError("Unable to resolve uploaded file path.")
    return str(resolved_path)


def _derived_paths(video_path: str, outputs_dir: str) -> dict[str, str]:
    clip_stem = Path(video_path).stem or "clip"
    outputs = Path(outputs_dir)
    return {
        "frames_json": str(outputs / f"{clip_stem}_frames.json"),
        "segments_csv": str(outputs / f"{clip_stem}_segments.csv"),
        "perception_json": str(outputs / f"{clip_stem}_perception.json"),
    }


def _set_paths(frames_json: str | None = None, segments_csv: str | None = None, perception_json: str | None = None) -> None:
    if frames_json is not None:
        st.session_state["frames_json_path"] = frames_json
    if segments_csv is not None:
        st.session_state["segments_csv_path"] = segments_csv
    if perception_json is not None:
        st.session_state["perception_json_path"] = perception_json


def _refresh_report_paths(video_path: str, outputs_dir: str) -> None:
    clip_stem = Path(video_path).stem or "clip"
    st.session_state["report_json_paths"] = services.list_report_files(outputs_dir, clip_stem, show_deprecated=False)

            ) or evidence_video_path

            report_segment = {}
            segments = report_payload.get("segments", [])
            if isinstance(segments, list) and segments and isinstance(segments[0], dict):
                report_segment = segments[0]
            report_segment_id = str(report_segment.get("segment_id", "")).strip()

            segment_options = [str(item.get("segment_id", f"segment_{idx+1}")) for idx, item in enumerate(perception_items)]
            default_segment_idx = segment_options.index(report_segment_id) if report_segment_id in segment_options else 0
            selected_segment_id = st.selectbox(
                "Segment",
                options=segment_options,
                index=default_segment_idx,
                key="evidence_segment_selector",
            )
            selected_perception = next(
                (item for item in perception_items if str(item.get("segment_id", "")) == selected_segment_id),
                perception_items[default_segment_idx],
            )

            claims = report_segment.get("claims", []) if isinstance(report_segment, dict) else []
            claims = [claim for claim in claims if isinstance(claim, dict)]
            claim_options = ["all"] + [str(claim.get("claim_id", "claim")) for claim in claims]
            selected_claim_id = st.selectbox("Claim Focus", options=claim_options, index=0, key="evidence_claim_focus")
            selected_claim = next((c for c in claims if str(c.get("claim_id", "")) == selected_claim_id), None)

            summary = services.extract_summary(report_payload)
            seg_start = float(selected_perception.get("start_sec", report_segment.get("start_sec", 0.0)))
            seg_end = float(selected_perception.get("end_sec", report_segment.get("end_sec", seg_start)))
            st.markdown('<div class="tir-card">', unsafe_allow_html=True)
            st.write(f"event_type: `{summary['event_type']}`")
            st.write(f"severity: `{summary['severity']}`")
            st.write(f"review_required: `{'Yes' if summary['review_required'] else 'No'}`")
            st.write(f"segment window: `{seg_start:.2f}s - {seg_end:.2f}s`")
            st.write(f"number_of_claims: `{summary['claims_count']}`")
            st.write(f"uncertainty flags: `{summary['uncertainty_flags']}`")
            st.markdown("</div>", unsafe_allow_html=True)

            st.markdown("**Claims & Evidence**")
            if claims:
                keyframes_for_segment = selected_perception.get("keyframes", [])
                keyframes_for_segment = [str(path) for path in keyframes_for_segment] if isinstance(keyframes_for_segment, list) else []
                tracks_csv_path = services.resolve_tracks_csv_for_report(report_path, source_video_from_report)

                for claim in claims:
                    claim_id = str(claim.get("claim_id", "claim"))
                    support_info = services.evaluate_claim_visual_support(
                        claim=claim,
                        keyframe_paths=keyframes_for_segment,
                        frames_json_path=evidence_frames_json if Path(evidence_frames_json).exists() else None,
                        video_path=source_video_from_report,
                        tracks_csv_path=tracks_csv_path,
                    )
                    header = (
                        f"{claim_id} | conf={float(claim.get('support_confidence', 0.0)):.2f} | "
                        f"visual={support_info['visual_support_level']}"
                    )
                    with st.expander(header, expanded=(selected_claim_id == claim_id)):
                        st.write(f"statement: {claim.get('statement', '')}")
                        evidence_frames = claim.get("evidence_frames", [])
                        if isinstance(evidence_frames, list) and evidence_frames:
                            st.write(f"evidence_frames: {evidence_frames}")
                        else:
                            st.write("evidence_frames: No explicit frame links")
                        st.write(f"uncertainty_flags: {claim.get('uncertainty_flags', [])}")
                        st.write(f"has_visual_boxes_any_evidence_frame: {support_info['has_visual_boxes_any_evidence_frame']}")
            else:
                st.info("No claims in selected report.")

            st.markdown("**Actors**")
            actors = report_payload.get("actors", [])
            if isinstance(actors, list) and actors:
                import pandas as pd

                actor_rows = []
                for actor in actors:
                    if isinstance(actor, dict):
                        actor_rows.append(
                            {
                                "actor_id": actor.get("actor_id", ""),
                                "role": actor.get("role", ""),
                                "description": actor.get("description", ""),
                            }
                        )
                st.dataframe(pd.DataFrame(actor_rows), use_container_width=True)
            else:
                st.info("No actors available.")

            st.markdown("**Audit & Provenance**")
            audit = report_payload.get("audit", {})
            notes = str(audit.get("notes", "")) if isinstance(audit, dict) else ""
            track_source = services.resolve_tracks_csv_for_report(report_path, source_video_from_report)
            track_exists = bool(track_source and Path(track_source).exists())
            track_mtime = ""
            if track_exists and track_source:
                track_mtime = datetime.fromtimestamp(Path(track_source).stat().st_mtime).strftime("%Y-%m-%d %H:%M:%S")

            kv_rows = []
            if isinstance(report_meta, dict):
                kv_rows.append({"key": "report_id", "value": str(report_meta.get("report_id", ""))})
                kv_rows.append({"key": "created_at", "value": str(report_meta.get("created_at", ""))})
            kv_rows.append({"key": "track_source", "value": str(track_source) if track_source else "not found"})
            kv_rows.append({"key": "track_exists", "value": "Yes" if track_exists else "No"})
            if track_mtime:
                kv_rows.append({"key": "track_last_modified", "value": track_mtime})
            primary_keys = {str(row["key"]).strip().lower() for row in kv_rows}
            for pair in _parse_audit_notes_pairs(notes):
                if str(pair.get("key", "")).strip().lower() in primary_keys:
                    continue
                kv_rows.append(pair)

            kv_html = '<table class="tir-kv">' + "".join(
                f"<tr><td>{row['key']}</td><td>{row['value']}</td></tr>" for row in kv_rows
            ) + "</table>"
            st.markdown(kv_html, unsafe_allow_html=True)

            st.markdown("**Human Review**")
            override_event_type_ev = st.selectbox(
                "override_event_type",
                options=["keep", "none", "near_miss", "collision", "obstruction", "violation"],
                index=0,
                key="evidence_override_event_type",
            )
            override_severity_ev = st.selectbox(
                "override_severity",
                options=["keep", "low", "medium", "high", "unknown"],
                index=0,
                key="evidence_override_severity",
            )
            reviewer_note_ev = st.text_area("reviewer_note", value="", height=90, key="evidence_reviewer_note")
            if st.button("Save Reviewed Copy", use_container_width=True, key="evidence_save_reviewed"):
                try:
                    reviewed_path = services.save_reviewed_copy(
                        report_path=report_path,
                        override_event_type=override_event_type_ev,
                        override_severity=override_severity_ev,
                        reviewer_note=reviewer_note_ev,
                    )
                    st.success(f"Reviewed copy saved: {reviewed_path}")
                except Exception as exc:
                    st.error(f"Failed to save reviewed copy: {exc}")

        with left_panel:
            st.markdown("### Incident Video")
            playback_mode = st.radio(
                "Playback mode",
                options=["Guided keyframes (recommended)", "Play full annotated clip"],
                index=0,
                key="evidence_playback_mode",
            )
            annotated_candidate = None
            detect_result = st.session_state.get("detecttrack_result", {})
            if detect_result.get("annotated_video") and Path(str(detect_result["annotated_video"])).exists():
                annotated_candidate = str(detect_result["annotated_video"])
            else:
                candidate_webm = Path(outputs_dir) / f"{evidence_video_stem}_annotated.webm"
                candidate_mp4 = Path(outputs_dir) / f"{evidence_video_stem}_annotated.mp4"
                if candidate_webm.exists():
                    annotated_candidate = str(candidate_webm)
                elif candidate_mp4.exists():
                    annotated_candidate = str(candidate_mp4)

            if playback_mode == "Play full annotated clip":
                st.info("Use this for context; decisions are based on evidence frames.")
            if annotated_candidate:
                candidate_path = Path(annotated_candidate)
                if candidate_path.stat().st_size <= 0:
                    st.warning("Annotated video file is empty. Re-run Detect+Track to regenerate it.")
                else:
                    _render_video(annotated_candidate)
            else:
                st.info("Annotated video not found. Run Detect+Track with annotated output enabled.")

            st.markdown("### Evidence Frame Inspector")
            keyframes_raw = selected_perception.get("keyframes", [])
            keyframes_all = [str(path) for path in keyframes_raw] if isinstance(keyframes_raw, list) else []
            existing_keyframes = [path for path in keyframes_all if Path(path).exists()]
            missing_count = len(keyframes_all) - len(existing_keyframes)
            if missing_count > 0:
                st.warning(f"{missing_count} keyframe files are missing and were skipped.")

            keyframes_filtered = (
                services.map_claim_to_keyframes(selected_claim, existing_keyframes) if selected_claim else existing_keyframes
            )
            if not keyframes_filtered and selected_claim:
                st.info("Selected claim has no direct keyframe links. Showing all segment keyframes.")
                keyframes_filtered = existing_keyframes

            if not keyframes_filtered:
                st.info("No keyframes available for this segment.")
            else:
                keyframe_meta_rows: list[tuple[str, str, dict[str, Any]]] = []
                for keyframe_path in keyframes_filtered:
                    meta = services.resolve_keyframe_metadata(
                        keyframe_path=keyframe_path,
                        frames_json_path=evidence_frames_json if Path(evidence_frames_json).exists() else None,
                        video_path=source_video_from_report,
                        fallback_fps=2.0,
                    )
                    sampled_idx = meta.get("sampled_frame_idx")
                    source_idx = meta.get("source_frame_idx")
                    timestamp = meta.get("timestamp_sec")
                    quality = meta.get("mapping_quality", "fallback")
                    label = (
                        f"kf={sampled_idx if sampled_idx is not None else 'na'} | "
                        f"src={source_idx if source_idx is not None else 'na'} | "
                        f"t={(float(timestamp) if timestamp is not None else -1):.2f}s | "
                        f"map={quality}"
                    )
                    keyframe_meta_rows.append((label, keyframe_path, meta))

                labels = [row[0] for row in keyframe_meta_rows]
                selected_label = st.selectbox("Keyframe", options=labels, index=0, key="evidence_keyframe_label")
                selected_row = next(row for row in keyframe_meta_rows if row[0] == selected_label)
                selected_keyframe_path = selected_row[1]
                selected_meta = selected_row[2]

                if str(selected_meta.get("mapping_quality", "fallback")) != "exact":
                    st.warning(f"Mapping quality is `{selected_meta.get('mapping_quality')}`; timestamp may be approximate.")

                track_source_path = services.resolve_tracks_csv_for_report(report_path, source_video_from_report)
                ts_sec = selected_meta.get("timestamp_sec")
                box_render_path = Path(outputs_dir) / "_rendered_frames" / (
                    f"{Path(selected_keyframe_path).stem}_track_overlay.jpg"
                )
                box_meta = services.render_keyframe_with_tracks(
                    keyframe_image_path=selected_keyframe_path,
                    keyframe_timestamp_sec=float(ts_sec) if ts_sec is not None else None,
                    tracks_csv_path=track_source_path,
                    out_path=str(box_render_path),
                    video_path=source_video_from_report,
                )

                top_claim = selected_claim
                if top_claim is None and claims:
                    top_claim = max(claims, key=lambda c: float(c.get("support_confidence", 0.0)))
                top_claim_id = str(top_claim.get("claim_id", "n/a")) if isinstance(top_claim, dict) else "n/a"
                top_event = str(top_claim.get("event_type", summary["event_type"])) if isinstance(top_claim, dict) else summary["event_type"]
                top_conf = float(top_claim.get("support_confidence", 0.0)) if isinstance(top_claim, dict) else 0.0

                overlay_lines = [
                    f"t={(float(ts_sec) if ts_sec is not None else -1):.2f}s",
                    f"segment={selected_segment_id}",
                    f"claim={top_claim_id}",
                    f"event={top_event} conf={top_conf:.2f}",
                ]
                overlay_path = services.render_annotated_keyframe(
                    image_path=str(box_render_path),
                    output_path="",
                    overlay_lines=overlay_lines,
                )
                st.image(overlay_path, use_container_width=True)
                st.caption(f"Evidence frame {labels.index(selected_label) + 1} of {len(labels)}")
                st.caption(f"Matched boxes: {box_meta['boxes_drawn']} (rows={box_meta['matched_rows']})")

                if selected_claim is not None:
                    support = services.evaluate_claim_visual_support(
                        claim=selected_claim,
                        keyframe_paths=existing_keyframes,
                        frames_json_path=evidence_frames_json if Path(evidence_frames_json).exists() else None,
                        video_path=source_video_from_report,
                        tracks_csv_path=track_source_path,
                    )
                    strongest = support.get("strongest_evidence_frame")
                    st.write(f"visual_support_level: `{support['visual_support_level']}`")
                    if (
                        str(selected_claim.get("event_type", "")) == "collision"
                        and strongest
                        and strongest != selected_keyframe_path
                    ):
                        st.info(
                            "Selected frame is context; inspect strongest evidence frame for peak event visibility."
                        )
                        if st.button("Jump to strongest evidence frame", key="jump_strongest_evidence"):
                            target = next(
                                (label for label, path, _meta in keyframe_meta_rows if path == strongest),
                                None,
                            )
                            if target is not None:
                                st.session_state["evidence_keyframe_label"] = target
                                st.rerun()
