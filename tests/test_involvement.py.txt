def test_normalize_involvement_response_filters_ids_and_fallback() -> None:
    normalized = normalize_involvement_response(
        payload={
            "primary_track_ids": [24, 999, "118"],
            "secondary_track_ids": [24, 100, "x"],
            "decision_confidence": 1.2,
            "rationale": "Clear impact pair.",
            "ambiguity_flags": ["occlusion", "occlusion", ""],
        },
        candidate_track_ids=[24, 100, 118],
        fallback_primary=[100, 118],
    )
    assert normalized["primary_track_ids"] == [24, 118]
    assert normalized["secondary_track_ids"] == [100]
    assert abs(normalized["decision_confidence"] - 1.0) < 1e-9
    assert normalized["ambiguity_flags"] == ["occlusion"]


def test_run_involvement_for_report_fallback_mode(tmp_path: Path) -> None:
    report_json = tmp_path / "report.json"
    perception_json = tmp_path / "perception.json"
    frames_json = tmp_path / "frames.json"
    tracks_csv = tmp_path / "tracks.csv"
    out_json = tmp_path / "involvement.json"

    report_json.write_text(
        json.dumps(
            {
                "meta": {"source_video": "data/raw_videos/demo.mp4"},
                "segments": [
                    {
                        "segment_id": "s_collision",
                        "claims": [
                            {
                                "claim_id": "clm_001",
                                "event_type": "collision",
                                "statement": "Collision observed.",
                                "support_confidence": 0.95,
                                "evidence_frames": [8, 10],
                                "uncertainty_flags": [],
                            }
                        ],
                    }
                ],
            }
        ),
        encoding="utf-8",
    )
    perception_json.write_text(
        json.dumps(
            [
                {
                    "segment_id": "s_collision",
                    "start_sec": 4.0,
                    "end_sec": 10.0,
                    "keyframes": [
                        "data/frame_cache/demo/f_000008.jpg",
                        "data/frame_cache/demo/f_000010.jpg",
                    ],
                }
            ]
        ),
        encoding="utf-8",
    )
    frames_json.write_text(
        json.dumps(
            {
                "frames": [
                    {"frame_id": 8, "orig_frame_idx": 120, "time_sec": 4.0, "path": "data/frame_cache/demo/f_000008.jpg"},
                    {"frame_id": 10, "orig_frame_idx": 150, "time_sec": 5.0, "path": "data/frame_cache/demo/f_000010.jpg"},
                ]
            }
        ),
        encoding="utf-8",
    )
    tracks_csv.write_text(
        "\n".join(
            [
                "frame_idx,timestamp_sec,track_id,class_id,class_name,conf,x1,y1,x2,y2",
                "120,4.00,100,2,car,0.91,10,10,60,60",
                "120,4.00,118,2,car,0.89,20,20,70,70",
                "150,5.00,100,2,car,0.93,11,11,61,61",
                "150,5.00,118,2,car,0.90,21,21,71,71",
            ]
        ),
        encoding="utf-8",
    )

    result = run_involvement_for_report(
        report_json_path=report_json,
        perception_json_path=perception_json,
        frames_json_path=frames_json,
        tracks_csv_path=tracks_csv,
        out_json_path=out_json,
        model="gpt-4.1-mini",
        use_ai=False,
    )
    assert result["claims"] == 1
    assert result["ai_used"] == 0
    assert out_json.exists()

    payload = json.loads(out_json.read_text(encoding="utf-8"))
    assert isinstance(payload.get("claims"), list)
    claim = payload["claims"][0]
    assert claim["method"] == "heuristic_fallback"
    assert len(claim["primary_track_ids"]) >= 1
    assert "grounded_statement" in claim
    assert "track" in str(claim["grounded_statement"]).lower()
    assert "motion context" in str(claim["grounded_statement"]).lower()

    perception_json = tmp_path / "perception.json"
    out_json = tmp_path / "involvement.json"

    report_json.write_text(
        json.dumps(
            {
                "meta": {"source_video": "data/raw_videos/demo.mp4"},
                "segments": [
                    {
                        "segment_id": "s1",
                        "claims": [
                            {
                                "claim_id": "clm_001",
                                "event_type": "collision",
                                "statement": "Collision claim A.",
                                "support_confidence": 0.95,
                                "evidence_frames": [9, 10, 11],
                                "uncertainty_flags": [],
                            },
                            {
                                "claim_id": "clm_002",
                                "event_type": "collision",
                                "statement": "Collision claim B.",
                                "support_confidence": 0.94,
                                "evidence_frames": [9, 10, 11],
                                "uncertainty_flags": [],
                            },
                        ],
                    }
                ],
            }
        ),
        encoding="utf-8",
    )
    perception_json.write_text(
        json.dumps(
            [
                {
                    "segment_id": "s1",
                    "start_sec": 4.3,
                    "end_sec": 8.3,
                    "keyframes": [
                        "data/frame_cache/demo/f_000009.jpg",
                        "data/frame_cache/demo/f_000011.jpg",

                }
            ]

        encoding="utf-8",
    )

    def _fake_resolve(claim: dict[str, Any], **_kwargs: Any) -> dict[str, Any]:
        claim_id = str(claim.get("claim_id", "")).strip()
        if claim_id == "clm_001":
            method = "ai"
            conf = 0.92
        else:
            method = "heuristic_fallback"
            conf = 0.60
        return {
            "claim_id": claim_id,
            "event_type": "collision",
            "candidate_track_ids": [21, 100, 118],
            "keyframes_used": ["data/frame_cache/demo/f_000009.jpg", "data/frame_cache/demo/f_000011.jpg"],
            "strongest_keyframe": "data/frame_cache/demo/f_000011.jpg",
            "impact_anchor_track_ids": [21, 118, 100],
            "impact_anchor_pair_hint": [21, 118],
            "impact_anchor_pair_score": 0.38,
            "impact_pair_candidates": [
                {"pair": [21, 118], "peak_timestamp_sec": 5.5, "peak_score": 0.38, "selection_score": 0.56},
                {"pair": [100, 118], "peak_timestamp_sec": 6.5, "peak_score": 0.34, "selection_score": 0.43},
            ],
            "method": method,
            "primary_track_ids": [21, 118],
            "secondary_track_ids": [100],
            "decision_confidence": conf,
            "rationale": "resolved",
            "ambiguity_flags": [],
            "grounded_statement": "primary collision statement",
            "tracking_notes": [],
        }

    monkeypatch.setattr("src.involvement.resolve_claim_involvement", _fake_resolve)

    result = run_involvement_for_report(
        report_json_path=report_json,
        perception_json_path=perception_json,
        frames_json_path=None,
        tracks_csv_path=None,
        out_json_path=out_json,
        model="gpt-4.1-mini",
