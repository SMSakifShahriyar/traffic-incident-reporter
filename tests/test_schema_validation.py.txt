from __future__ import annotations

import json
import sys
from pathlib import Path

import pytest
from pydantic import ValidationError


PROJECT_ROOT = Path(__file__).resolve().parents[1]
if str(PROJECT_ROOT) not in sys.path:
    sys.path.insert(0, str(PROJECT_ROOT))

from src.reporter import load_report
from src.schema import IncidentReport


SAMPLE_REPORT = PROJECT_ROOT / "docs" / "sample_report.json"


def test_sample_report_validates() -> None:
    report = load_report(SAMPLE_REPORT)
    assert report.overall_assessment.event_type.value == "near_miss"


def test_invalid_report_fails_validation() -> None:
    payload = json.loads(SAMPLE_REPORT.read_text(encoding="utf-8"))
    payload["segments"][0]["start_sec"] = 9.0
    payload["segments"][0]["end_sec"] = 7.0
    payload["segments"][0]["evidence_frames"] = []
    payload["segments"][0]["segment_confidence"] = 1.2

    with pytest.raises(ValidationError):
        IncidentReport.model_validate(payload)
