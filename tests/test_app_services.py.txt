def test_should_reuse_perception_true_and_false(tmp_path: Path) -> None:
    perception_json = tmp_path / "demo_perception.json"
    perception_json.write_text("[]", encoding="utf-8")
    meta_path = Path(f"{perception_json}.meta.json")

    video_path = tmp_path / "demo.mp4"
    video_path.write_bytes(b"fake")
    segments_csv = tmp_path / "demo_segments.csv"
    segments_csv.write_text("segment_id,start_sec,end_sec\ns1,0,2\n", encoding="utf-8")

    matching_meta = {
        "video_path": str(video_path.resolve()),
        "segments_csv": str(segments_csv.resolve()),
        "model": "gpt-4.1-mini",
        "max_frames": 4,
    }
    meta_path.write_text(json.dumps(matching_meta), encoding="utf-8")

    assert app_services.should_reuse_perception(
        perception_json=perception_json,
        video_path=video_path,
        segments_csv=segments_csv,
        model="gpt-4.1-mini",
        max_frames=4,
    )

    assert not app_services.should_reuse_perception(
        perception_json=perception_json,
        video_path=video_path,
        segments_csv=segments_csv,
        model="gpt-4.1-mini",
        max_frames=5,
    )

def test_render_annotated_keyframe_cache_reuse(tmp_path: Path, monkeypatch) -> None:
    monkeypatch.chdir(tmp_path)
    app_services.ensure_dirs()

    src_image = tmp_path / "source_cache.jpg"
    img = np.full((70, 110, 3), 120, dtype=np.uint8)
    assert cv2.imwrite(str(src_image), img)

    first = app_services.render_annotated_keyframe(
        image_path=str(src_image),
        output_path="ignored.jpg",
        overlay_lines=["t=2.00s", "segment=s2"],
    )
    first_path = Path(first)
    first_mtime = first_path.stat().st_mtime_ns

    second = app_services.render_annotated_keyframe(
        image_path=str(src_image),
        output_path="ignored.jpg",
        overlay_lines=["t=2.00s", "segment=s2"],
    )
    second_path = Path(second)
    second_mtime = second_path.stat().st_mtime_ns

    assert first == second
    assert first_mtime == second_mtime

        output_path="ignored.jpg",

    )

        image_path=str(src_image),
        output_path="ignored.jpg",
        overlay_lines=["t=2.00s", "segment=s2"],
    )

    assert first != second

def test_resolve_keyframe_metadata_exact_and_fallback(tmp_path: Path, monkeypatch) -> None:
    monkeypatch.chdir(tmp_path)
    app_services.ensure_dirs()

    keyframe = tmp_path / "data" / "frame_cache" / "demo" / "f_000004.jpg"
    keyframe.parent.mkdir(parents=True, exist_ok=True)
    img = np.full((48, 64, 3), 100, dtype=np.uint8)
    assert cv2.imwrite(str(keyframe), img)

    frames_json = tmp_path / "data" / "outputs" / "demo_frames.json"
    frames_json.parent.mkdir(parents=True, exist_ok=True)
    frames_json.write_text(
        json.dumps(
            {
                "video_path": "data/raw_videos/demo.mp4",
                "probe": {"fps": 30.0},
                "frames": [
                    {
                        "frame_id": 4,
                        "orig_frame_idx": 60,
                        "time_sec": 2.0,
                        "path": str(keyframe),
                        "width": 64,
                        "height": 48,
                    }
                ],
            }
        ),
        encoding="utf-8",
    )

    exact = app_services.resolve_keyframe_metadata(
        keyframe_path=str(keyframe),
        frames_json_path=str(frames_json),
        video_path=None,
    )
    assert exact["mapping_quality"] == "exact"
    assert exact["sampled_frame_idx"] == 4
    assert exact["source_frame_idx"] == 60
    assert abs(float(exact["timestamp_sec"]) - 2.0) < 1e-9

    fallback = app_services.resolve_keyframe_metadata(
        keyframe_path="data/frame_cache/demo/f_000010.jpg",
        frames_json_path=None,
        video_path=None,
    )
    assert fallback["mapping_quality"] == "fallback"
    assert fallback["sampled_frame_idx"] == 10
    assert fallback["source_frame_idx"] is None
    assert abs(float(fallback["timestamp_sec"]) - 5.0) < 1e-9

    monkeypatch.chdir(tmp_path)
    app_services.ensure_dirs()

    video = tmp_path / "data" / "raw_videos" / "demo.mp4"

    video.write_bytes(b"fake")

    outputs = tmp_path / "data" / "outputs"
    outputs.mkdir(parents=True, exist_ok=True)
    reports_dir = outputs / "reports"
    reports_dir.mkdir(parents=True, exist_ok=True)
    primary_report = reports_dir / "demo_s1_collision_report.json"
    primary_report.write_text(
        json.dumps(
            {
                "segments": [
                    {
                        "segment_id": "s1",
                        "claims": [{"claim_id": "clm_001", "event_type": "collision", "support_confidence": 0.9}],
                    }
                ]
            }
        ),
        encoding="utf-8",
    )

    tracks_csv = outputs / "demo_tracks.csv"
    tracks_csv.write_text("frame_idx,timestamp_sec,track_id,class_id,class_name,conf,x1,y1,x2,y2\n", encoding="utf-8")

        app_services,
        "run_ingest_segment",
        lambda **_kwargs: {
            "frames_json": str(outputs / "demo_frames.json"),
            "segments_csv": str(outputs / "demo_segments.csv"),
            "segment_count": 2,
        },
    )
    monkeypatch.setattr(
        app_services,
        "run_detecttrack",
        lambda **_kwargs: {
            "tracks_csv": str(tracks_csv),
            "summary_json": str(outputs / "demo_tracks_summary.json"),
            "annotated_video": str(outputs / "demo_annotated.webm"),
            "num_tracks": 5,
            "num_detections": 100,
        },
    )
    monkeypatch.setattr(
        app_services,
        "run_perceive",
        lambda **_kwargs: {
            "perception_json": str(outputs / "demo_perception.json"),
            "ok": 2,
            "failed": 0,
            "reused_cache": False,
        },
    )
    monkeypatch.setattr(
        app_services,

        lambda **_kwargs: {
            "report_paths": [str(primary_report)],

            "failed": 0,
        },
